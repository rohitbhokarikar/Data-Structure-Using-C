Programme for Bubble Sort -

#include <stdio.h>

int main()
{
int array[100], n, i, r, swap;

printf("Enter number of elements\n");
scanf("%d", &n);
system("color 0");

printf("Enter %d elements \n", n);

for (i = 0; i < n; i++)
scanf("%d", &array[i]);

for (i = 0 ; i < ( n - 1 ); i++)
{
for (r = 0 ; r < n - i - 1; r++)
{
if (array[r] > array[r+1])
{
swap = array[r];
array[r] = array[r+1];
array[r+1] = swap;
}
}
}

printf("Sorted list in ascending order:\n");

for ( i = 0 ; i< n ; i++ )
printf("%d\n", array[i]);

return 0;
}
